while (iter.hasNext()) {
    var elem = iter.next();
    console.log(elem);
}

list.removeAt(0);

var iter2 = list.iterator(2);
while (iter2.hasNext()) {
    var elem = iter2.next();
    if (elem === 3) {
        iter2.remove();
        console.log(elem + ' removed');
    } else if (elem === 5) {
        iter2.set(0);
        console.log(elem + ' changed: 5 -> 0');
    } else {
        console.log(elem);
    }
}

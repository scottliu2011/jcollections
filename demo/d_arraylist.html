//添加几个元素
list.add('hello');
list.add('world');
list.add('scott');

console.log('after add: ' + list);//[hello, world, scott, 1, 2, 3]
console.log('get 0: ' + list.get(0));//获取索引为0的元素，即hello
console.log('contains soctt? ' + list.contains('scott'));//查看是否包含scott
console.log('indexOf soctt? ' + list.indexOf('scott'));//查看scott在集合中的索引
$
list.insert(3, -1, 0);//在索引为3的位置插入两个元素：-1和0
console.log('after insert: ' + list);//[hello, world, scott, -1, 0, 1, 2, 3]
list.removeAt(2);//移除索引为2对应的元素
console.log('after removeAt: ' + list);//[hello, world, -1, 0, 1, 2, 3]
list.removeElement('world');//移除值为world的元素
console.log('after removeElement: ' + list);//[hello, -1, 0, 1, 2, 3]
$
list.set(1, 1);//设置索引为1的元素值为1
console.log('after set: ' + list);//[hello, 1, 0, 1, 2, 3]
console.log('indexOf 1? ' + list.indexOf(1));//查看元素值为1第一次出现在集合中的索引值
console.log('lastIndexOf 1? ' + list.lastIndexOf(1));//查看元素值为1最后一次出现在集合中的索引值
list.removeRange(0, 2);//移除从0到2之间的元素（包括0的位置，不包括2的位置）
console.log('after removeRange: ' + list);//[0, 1, 2, 3]
$
list.clear();//清空集合

console.log('size: ' + list.size());//查看集合元素个数

var list2 = new ArrayList([1, 2, 3]);//创建新的ArrayList实例
list.addAll(list2);//把新创建的实例元素全部添加到list实例中
var list3 = new LinkedList();//创建新的LinkedList实例
list3.add(4, 5, 6);//向LinkedList实例中添加多个元素
list.addAll(list3);//把新LinkedList实例元素全部添加到list实例中
console.log('after addAll: ' + list);//[1, 2, 3, 4, 5, 6]
var set = new HashSet();//新建一个HashSet实例
set.add('hello');//向HashSet实例中添加一个元素
list.insertAll(0, set);//把HashSet实例元素插入到list实例中的索引为0处
console.log('after insertAll: ' + list);//[hello, 1, 2, 3, 4, 5, 6]

var iter = list.iterator();//创建一个list迭代器
while (iter.hasNext()) {//条件：是否还有下一个元素
	var elem = iter.next();//获取下一个元素
	if (elem === 6) {//如果元素值等于6，就从list集合中移除
		iter.remove();//remove 6
	}
	if (elem === 'hello') {//如果元素值等于hello，就设置成新值0
		iter.set(0);//'hello' => 0
	}
}
console.log('after iterator: ' + list);//[0, 1, 2, 3, 4, 5]
